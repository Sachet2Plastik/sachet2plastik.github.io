194 void update(float deltaTimeSecond) {
195 if (_isPlaying && !_currentAnimation.empty()) {
196 _elapsedTime += deltaTimeSecond;
197 const auto& anime = _animations[_currentAnimation];
198
199 if (_elapsedTime >= anime.frameDuration) {
200 _elapsedTime -= anime.frameDuration;
201 _currentFrame++;
202
203 if (_currentFrame >= anime.frames.size()) {
204 if (!anime.loop) {
205 _isPlaying = false;
206 }
207
208 _currentFrame = anime.loop ? 0 : anime.frames.size() - 1;
209 }
210 setTextureRect(anime.frames[_currentFrame]);
211 }
212 }
213 }
214
220 void play(const std::string& animation_name) {
221 auto it = _animations.find(animation_name);
222
223 if (it != _animations.end()) {
224 _currentAnimation = animation_name;
225 _currentFrame = 0;
226 _elapsedTime = 0.0f;
227 _isPlaying = true;
228 setTextureRect(_animations[animation_name].frames[_currentFrame]);
229 } else {
230 LOG(util::WARNING, "Animation not found: %s", animation_name.c_str());
231 }
232 }
