187 void update(float deltaTimeSecond) {
188 if (_isPlaying && !_currentAnimation.empty()) {
189 _elapsedTime += deltaTimeSecond;
190 const auto& anime = _animations[_currentAnimation];
191
192 if (_elapsedTime >= anime.frameDuration) {
193 _elapsedTime -= anime.frameDuration;
194 _currentFrame++;
195
196 if (_currentFrame >= anime.frames.size()) {
197 if (!anime.loop) {
198 _isPlaying = false;
199 }
200
201 _currentFrame = anime.loop ? 0 : anime.frames.size() - 1;
202 }
203 setTextureRect(anime.frames[_currentFrame]);
204 }
