255template <class Component>
256SparseArray<Component>& registry::register_component()
257{
258 auto it = _componentsArrays.find(typeid(Component));
259
260 if (it == _componentsArrays.end()) {
261 _componentsArrays[typeid(Component)] = SparseArray<Component>();
262
263 _eraseFunctions[typeid(Component)] = [](registry& reg, entity_t entity) {
264 SparseArray<Component>& set = reg.get_components<Component>();
265 set.erase(entity);
266 };
267 } else {
268 boost::core::scoped_demangled_name demangled(typeid(Component).name());
269 LOG(util::WARNING, "Component %s already registered", demangled.get());
270 }
271
272 return std::any_cast<SparseArray<Component>&>(_componentsArrays[std::type_index(typeid(Component))]);
273}

360template <typename... Component, typename Function>
361void registry::add_system(SystemHierarchy hierarchy, Function &&f)
362{
363 switch (hierarchy)
364 {
365 case SETUP:
366 _startupSystems.emplace_back(
367 [system = std::forward<Function>(f)](registry &r) {
368 system(r, r.get_components<Component>()...);
369 });
370 break;
371 case RUNNING:
372 _systems.emplace_back(
373 [system = std::forward<Function>(f)](registry &r) {
374 system(r, r.get_components<Component>()...);
375 });
376 break;
377 case CLOSING:
378 _shutdownSystems.emplace_back(
379 [system = std::forward<Function>(f)](registry &r) {
380 system(r, r.get_components<Component>()...);
381 });
382 break;
383
384 }
