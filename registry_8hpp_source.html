384template <class Component>
385SparseArray<Component>& registry::register_component()
386{
387 auto it = _componentsArrays.find(typeid(Component));
388
389 if (it == _componentsArrays.end()) {
390 _componentsArrays[typeid(Component)] = SparseArray<Component>();
391
392 _eraseFunctions[typeid(Component)] = [](registry& reg, entity_t entity) {
393 SparseArray<Component>& set = reg.get_components<Component>();
394 set.erase(entity);
395 };
396 } else {
397 boost::core::scoped_demangled_name demangled(typeid(Component).name());
398 LOG(util::WARNING, "Component %s already registered", demangled.get());
399 }
400
401 return std::any_cast<SparseArray<Component>&>(_componentsArrays[std::type_index(typeid(Component))]);
402}

480void registry::remove_component(entity_t const &from)
481{
482 if (_activeIDs.find(from) == _activeIDs.end()) {
483 throw std::invalid_argument("Entity does not exist");
484 }
485
486 auto it = _eraseFunctions.find(typeid(Component));

489 it->second(*this, from);

493template <typename... Component, typename Function>
494void registry::add_system(SystemHierarchy hierarchy, Function &&f)
495{
496 switch (hierarchy)
497 {
498 case SETUP:
499 _startupSystems.emplace_back(
500 [system = std::forward<Function>(f)](registry &r) {
501 system(r, r.get_components<Component>()...);
502 });
503 break;
504 case RUNNING:
505 _systems.emplace_back(
506 [system = std::forward<Function>(f)](registry &r) {
507 system(r, r.get_components<Component>()...);
508 });
509 break;
510 case CLOSING:
511 _shutdownSystems.emplace_back(
512 [system = std::forward<Function>(f)](registry &r) {
513 system(r, r.get_components<Component>()...);
514 });
515 break;
516
517 }
518}
