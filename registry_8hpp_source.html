268template <class Component>
269SparseArray<Component>& registry::register_component()
270{
271 auto it = _componentsArrays.find(typeid(Component));
272
273 if (it == _componentsArrays.end()) {
274 _componentsArrays[typeid(Component)] = SparseArray<Component>();
275
276 _eraseFunctions[typeid(Component)] = [](registry& reg, entity_t entity) {
277 SparseArray<Component>& set = reg.get_components<Component>();
278 set.erase(entity);
279 };
280 }
281
282 return std::any_cast<SparseArray<Component>&>(_componentsArrays[std::type_index(typeid(Component))]);
283}

358{

360 throw std::invalid_argument("Entity does not exist");

370template <typename... Component, typename Function>
371void registry::add_system(SystemHierarchy hierarchy, Function &&f)
372{
373 switch (hierarchy)
374 {
375 case SETUP:
376 _startupSystems.emplace_back(
377 [system = std::forward<Function>(f)](registry &r) {
378 system(r, r.get_components<Component>()...);
379 });
380 break;
381 case RUNNING:
382 _systems.emplace_back(
383 [system = std::forward<Function>(f)](registry &r) {
384 system(r, r.get_components<Component>()...);
385 });
386 break;
387 case CLOSING:
388 _shutdownSystems.emplace_back(
389 [system = std::forward<Function>(f)](registry &r) {
390 system(r, r.get_components<Component>()...);
391 });
392 break;
393
394 }
