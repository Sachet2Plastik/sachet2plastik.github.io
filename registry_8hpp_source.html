330template <class Component>
331SparseArray<Component>& registry::register_component()
332{
333 auto it = _componentsArrays.find(typeid(Component));
334
335 if (it == _componentsArrays.end()) {
336 _componentsArrays[typeid(Component)] = SparseArray<Component>();
337
338 _eraseFunctions[typeid(Component)] = [](registry& reg, entity_t entity) {
339 SparseArray<Component>& set = reg.get_components<Component>();
340 set.erase(entity);
341 };
342 } else {
343 boost::core::scoped_demangled_name demangled(typeid(Component).name());
344 LOG(util::WARNING, "Component %s already registered", demangled.get());
345 }
346
347 return std::any_cast<SparseArray<Component>&>(_componentsArrays[std::type_index(typeid(Component))]);
348}

425 throw std::invalid_argument("Entity does not exist");

435template <typename... Component, typename Function>
436void registry::add_system(SystemHierarchy hierarchy, Function &&f)
437{
438 switch (hierarchy)
439 {
440 case SETUP:
441 _startupSystems.emplace_back(
442 [system = std::forward<Function>(f)](registry &r) {
443 system(r, r.get_components<Component>()...);
444 });
445 break;
446 case RUNNING:
447 _systems.emplace_back(
448 [system = std::forward<Function>(f)](registry &r) {
449 system(r, r.get_components<Component>()...);
450 });
451 break;
452 case CLOSING:
453 _shutdownSystems.emplace_back(
454 [system = std::forward<Function>(f)](registry &r) {
455 system(r, r.get_components<Component>()...);
456 });
457 break;
458
459 }
