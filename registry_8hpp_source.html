328template <class Component>
329SparseArray<Component>& registry::register_component()
330{
331 auto it = _componentsArrays.find(typeid(Component));
332
333 if (it == _componentsArrays.end()) {
334 _componentsArrays[typeid(Component)] = SparseArray<Component>();
335
336 _eraseFunctions[typeid(Component)] = [](registry& reg, entity_t entity) {
337 SparseArray<Component>& set = reg.get_components<Component>();
338 set.erase(entity);
339 };
340 } else {
341 boost::core::scoped_demangled_name demangled(typeid(Component).name());
342 LOG(util::WARNING, "Component %s already registered", demangled.get());
343 }
344
345 return std::any_cast<SparseArray<Component>&>(_componentsArrays[std::type_index(typeid(Component))]);
346}

423 throw std::invalid_argument("Entity does not exist");

433template <typename... Component, typename Function>
434void registry::add_system(SystemHierarchy hierarchy, Function &&f)
435{
436 switch (hierarchy)
437 {
438 case SETUP:
439 _startupSystems.emplace_back(
440 [system = std::forward<Function>(f)](registry &r) {
441 system(r, r.get_components<Component>()...);
442 });
443 break;
444 case RUNNING:
445 _systems.emplace_back(
446 [system = std::forward<Function>(f)](registry &r) {
447 system(r, r.get_components<Component>()...);
448 });
449 break;
450 case CLOSING:
451 _shutdownSystems.emplace_back(
452 [system = std::forward<Function>(f)](registry &r) {
453 system(r, r.get_components<Component>()...);
454 });
455 break;
456
457 }
458}
